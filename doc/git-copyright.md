# GIT Copyright check

```bash
for filename in `git ls-files | grep -v '^\.git/' | grep -v '^\.idea/' | grep -v '^\.DS_Store' | grep -v '^\.vscode/' | grep -v '^\.github/' | grep -v '^\.gitignore' | grep -v '^\.gitattributes'`
do 
  created_date="$(git log --follow --format=%ad --date=iso $filename | tail -1)"
  zzz="$(grep "SPDX-FileCopyrightText:" $filename | head -1 | sed 's/.*SPDX-FileCopyrightText: //')"
  printf "%s %s => %s\n" "$created_date" "$filename" "$zzz"
done | sort
```
